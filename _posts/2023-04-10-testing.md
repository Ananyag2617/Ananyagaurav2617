<script>
function analyze() {
  const input = document.getElementById("input").value;
  const url = `https://edamam-edamam-nutrition-analysis.p.rapidapi.com/api/nutrition-data?ingr=${input}`;
  const headers = {
    "X-RapidAPI-Host": "edamam-edamam-nutrition-analysis.p.rapidapi.com",
    "X-RapidAPI-Key": "afae82f074msh729fe72b128081fp1f40f4jsn36bd559b1ce4"
  };

  // Save the ingredients in a table
  const ingredientsTable = document.getElementById("ingredients-table");
  const row = ingredientsTable.insertRow();
  const cell = row.insertCell();
  cell.innerHTML = input;

  fetch(url, {
    method: "GET",
    headers: headers
  })
    .then(response => response.json())
    .then(data => {
      const nutrients = data.totalNutrients;
      const table = document.createElement("table");
      table.innerHTML = `
        <tr>
          <th>Nutrient</th>
          <th>Quantity</th>
          <th>Unit</th>
        </tr>
      `;
      for (let nutrient in nutrients) {
        const row = document.createElement("tr");
        row.innerHTML = `
          <td>${nutrients[nutrient].label}</td>
          <td>${nutrients[nutrient].quantity.toFixed(2)}</td>
          <td>${nutrients[nutrient].unit}</td>
        `;
        table.appendChild(row);
      }
      const result = document.getElementById("result");
      result.innerHTML = "";
      result.appendChild(table);
    });
}
</script>

<script>
    function addFoodItem(event) {
        event.preventDefault();
        const foodItemInput = document.getElementById("food-item");
        const caloriesInput = document.getElementById("calories");
        const foodTableBody = document.getElementById("food-table-body");
        const foodItem = foodItemInput.value;
        const calories = caloriesInput.value;
        const row = document.createElement("tr");
        const foodItemCell = document.createElement("td");
        const caloriesCell = document.createElement("td");
        const deleteButton = document.createElement("button");

        foodItemCell.appendChild(document.createTextNode(foodItem));
        caloriesCell.appendChild(document.createTextNode(calories));
        deleteButton.appendChild(document.createTextNode("Delete"));
        deleteButton.onclick = function () {
            if (confirm("Yes delete this")) {
                row.remove();
            }
        };
        row.appendChild(foodItemCell);
        row.appendChild(caloriesCell);
        row.appendChild(deleteButton);
        foodTableBody.appendChild(row);
        foodItemInput.value = "";
        caloriesInput.value = "";
    }
</script>

<script>
function calculateCalories() {
  // get user input
  var age = parseInt(document.getElementById("age").value);
  var caloriesConsumed = parseInt(document.getElementById("calories").value);

  // calculate recommended daily calorie intake based on age
  var recommendedCalories = 0;
  if (age < 3) {
    recommendedCalories = 1000;
  } else if (age < 9) {
    recommendedCalories = 1400;
  } else if (age < 13) {
    recommendedCalories = 1800;
  } else if (age < 19) {
    recommendedCalories = 2400;
  } else if (age < 31) {
    recommendedCalories = 2200;
  } else if (age < 51) {
    recommendedCalories = 2000;
  } else if (age < 71) {
    recommendedCalories = 1800;
  } else {
    recommendedCalories = 1600;
  }

  // calculate remaining calories needed
  var remainingCalories = recommendedCalories - caloriesConsumed;

  // output result
  var output = document.getElementById("output");
  output.innerHTML = "Based on your age, you should consume " + recommendedCalories + " calories per day. You have consumed " + caloriesConsumed + " calories today. You still need to consume " + remainingCalories + " calories to meet your daily requirement.";
}
</script>
